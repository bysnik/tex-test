%парсер тхт файлов
\newread\myfile
\def\readline#1#2{%
  \openin\myfile=#2\relax
  \readlinehelper{#1}{1}%
  \closein\myfile
}
\def\readlinehelper#1#2{%
  \ifeof\myfile
    % Если дошли до конца — ничего не делаем
  \else
    \read\myfile to \tmp
    \ifnum#2=#1\relax
      \tmp % выводим нужную строку
    \else
      \expandafter\readlinehelper\expandafter#1\expandafter{\number\numexpr#2+1\relax}%
    \fi
  \fi
}
